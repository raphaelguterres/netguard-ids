# NetGuard Endpoint Agent — Windows version metadata.
#
# Embedded into dist/agent.exe by PyInstaller via agent.spec
# (version='version_info.txt'). Without this file the .exe shows
# blank file properties in Explorer, which trips heuristics on
# several enterprise antivirus / EDR products.
#
# To bump the version, change agent/__init__.py and re-run
# scripts/bump_version_info.py (or edit the 4-tuple here by hand).
#
# Format reference:
#   https://pyinstaller.org/en/stable/usage.html#capturing-windows-version-data
VSVersionInfo(
  ffi=FixedFileInfo(
    # filevers / prodvers — 4-tuple of 16-bit ints. Keep the first
    # three in sync with __version__ in agent/__init__.py
    # (major, minor, patch, build). Build defaults to 0.
    filevers=(1, 0, 0, 0),
    prodvers=(1, 0, 0, 0),
    mask=0x3F,
    flags=0x0,
    OS=0x40004,
    fileType=0x1,
    subtype=0x0,
    date=(0, 0)
  ),
  kids=[
    StringFileInfo(
      [
        StringTable(
          # 040904B0 = US English, Unicode. Keep this — Windows
          # localised file properties dialogs read this code page.
          u'040904B0',
          [
            StringStruct(u'CompanyName', u'NetGuard'),
            StringStruct(u'FileDescription', u'NetGuard Endpoint Agent'),
            StringStruct(u'FileVersion', u'1.0.0.0'),
            StringStruct(u'InternalName', u'agent'),
            StringStruct(u'LegalCopyright', u'© NetGuard. All rights reserved.'),
            StringStruct(u'OriginalFilename', u'agent.exe'),
            StringStruct(u'ProductName', u'NetGuard Endpoint Agent'),
            StringStruct(u'ProductVersion', u'1.0.0.0'),
            StringStruct(u'Comments', u'Lightweight endpoint telemetry agent. https://github.com/raphaelguterres/netguard-ids'),
          ]
        )
      ]
    ),
    VarFileInfo([VarStruct(u'Translation', [0x0409, 1200])])
  ]
)
